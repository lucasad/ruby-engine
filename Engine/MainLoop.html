def run
    @last = SDL2::get_ticks
    @updates = 0

    loop do
        handle_events
        @updates += since_last
        for _ in (0..@updates/Quantum) do
            @game.tick Quantum
        end

        @renderer.render @game if @updates > 10
        @updates %= Quantum

        sleep 1.0/1000
    end
end
